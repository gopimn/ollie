  tmpfs /tmp tmpfs defaults,nosuid,noatime,nodev,noexec
  tmpfs /var/log tmpfs defaults,nosuid,noatime,nodev,noexec,size=64m

  pcm.playback {
      type plug
      slave.pcm "hw:CARD=ALSA,DEV=0"
  }
